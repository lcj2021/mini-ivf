149  {
150  std::reference_wrapper<ImplementationT> refWrapper(servantObj);
151 
152  RcfClientPtr rcfClientPtr = createServerStub(
153  (InterfaceT *) 0,
154  (ImplementationT *) 0, refWrapper);

156  return bindImpl(
157  name.empty() ?

159  name ,
160  rcfClientPtr);
